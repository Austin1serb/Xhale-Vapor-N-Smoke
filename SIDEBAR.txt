SIDEBAR

import React, { useState } from 'react';
import {
    AppBar,
    Toolbar,
    Typography,
    Container,
    Grid,
    Paper,
    IconButton,
    Box,
} from '@mui/material';
import { FcTodoList, FcBusinessman, FcComboChart } from 'react-icons/fc';
import AccountCircle from '@mui/icons-material/AccountCircle';
import MenuIcon from '@mui/icons-material/Menu';
import ProductList from '../components/ProductList';
import UserList from '../components/UserList';
import OrderList from '../components/OrderList';
import { Sidebar, Menu, MenuItem, SubMenu } from 'react-pro-sidebar';
import MenuOutlinedIcon from "@mui/icons-material/MenuOutlined";

const AdminDashboard = () => {
    const [sidebarCollapsed, setSidebarCollapsed] = useState(true); // Start with sidebar collapsed
    const [selectedComponent, setSelectedComponent] = useState(null);

    const handleSidebarToggle = () => {
        setSidebarCollapsed(!sidebarCollapsed);
    };

    const handleSidebarItemClick = (component) => {
        setSelectedComponent(component);
    };

    return (
        <div>
            <AppBar position="static">
                <Toolbar sx={{ justifyContent: 'space-between' }}>
                    <IconButton
                        edge="start"
                        color="inherit"
                        aria-label="menu"
                        onClick={handleSidebarToggle}
                    >

                    </IconButton>
                    <Typography variant="h6">Admin Dashboard</Typography>
                    <Box component="a" href="#" className="nav-link">
                        <IconButton
                            size="large"
                            edge="end"
                            aria-label="account of current user"
                            color="inherit"
                        >
                            <AccountCircle sx={{ fontSize: '32px' }} />
                        </IconButton>
                    </Box>
                </Toolbar>
            </AppBar>
            <Container>
                <Grid container spacing={3}>
                    <Grid item xs={12}>
                        <Paper>
                            {selectedComponent === 'productList' && <ProductList />}
                            {selectedComponent === 'userList' && <UserList />}
                            {selectedComponent === 'orderList' && <OrderList />}
                        </Paper>
                    </Grid>
                </Grid>
            </Container>
            <Sidebar collapsed={sidebarCollapsed}>

                <Menu iconShape="square">
                    <MenuItem>

                        <MenuOutlinedIcon
                            edge="end"
                            color="inherit"
                            aria-label="menu"
                            onClick={handleSidebarToggle} />

                    </MenuItem>
                    <MenuItem
                        icon={<FcTodoList />}
                        onClick={() => handleSidebarItemClick('productList')}
                    >
                        Product List
                    </MenuItem>
                    <MenuItem
                        icon={<FcBusinessman />}
                        onClick={() => handleSidebarItemClick('userList')}
                    >
                        User List
                    </MenuItem>
                    <MenuItem
                        icon={<FcComboChart />}
                        onClick={() => handleSidebarItemClick('orderList')}
                    >
                        Order List
                    </MenuItem>
                </Menu>
            </Sidebar>
        </div>
    );
};

export default AdminDashboard;
